% Document header ---------------------------------------------------------------
%   This template is based on "scrreprt" from the koma-script.
% ------------------------------------------------------------------------------
\documentclass[
    11pt, % font size
    DIV=10,
    ngerman, % for german umlauts, hyphenation, ...
    a4paper, % paper format
    oneside, % one sided document
    titlepage, % use a tile page
    parskip=half, % gap between paragraphs (half a line)
    headings=normal, % decrease size of headings
    listof=totoc, % list directories in the table of contents
    bibliography=totoc, % list bibliography in table of contents
    index=totoc, % list index in table of contents
    captions=tableheading, % label tables at bottom of table
    draft % state of the document (final/draft)
]{scrreprt}

% meta information -----------------------------------------------------------
%   Infos about the documents, like title, author, matrikelnr, etc
%   are defined in the file Meta.tex and are available globally for usage
% ------------------------------------------------------------------------------
\input{Meta}

% required packages -----------------------------------------------------------
%   required latex packages are listed in Packages.tex to keep this template
%   as clean as possible.
% ------------------------------------------------------------------------------
\input{Packages}

% Activate the creation of an index and the list of abbreviations ---------------
\makeindex
\makenomenclature

% Header, footer, margin, ... ---------------------------------------
\input{PageStyle}

% personal definitions for hyphenation
\include{Hyphenation}

% custom latex commands
\include{Commands}

% the actual document
%   the actual content of the thesis starts here.
%   single pages and chapters get outsources in their own files
%   which then get included here
% ------------------------------------------------------------------------------
\begin{document}

% also enumerate subsubsections
\setcounter{secnumdepth}{3}
\setcounter{tocdepth}{3}

% cover page and abstract without pagenumber
\ofoot{}
\include{CoverPage}
\include{Content/Abstract}
\ofoot{\pagemark}

% page numeration -----------------------------------------------------------
%   Before the main part, number the pages in big roman numerals
% ------------------------------------------------------------------------------
\pagenumbering{Roman}
\tableofcontents

% Abk�rzungsverzeichnis --------------------------------------------------------
\input{Content/Glossar}
% f�r korrekte �berschrift in der Kopfzeile
\clearpage\markboth{\nomname}{\nomname} 
\printnomenclature
\label{sec:Glossar}

\listoffigures % list of figures
\listoftables % list of tables
\renewcommand{\lstlistlistingname}{Verzeichnis der Listings}
\lstlistoflistings % Listings-Verzeichnis

% normal page numbers in the main part ------------------------------------------
\clearpage
\pagenumbering{arabic}

% include the content chapters inside "Content.tex" -------------------------
\input{Content}

% bibliography ---------------------------------------------------------
%   the bibliography gets build from the bixtex database in ""Bibliography.bib"
% ------------------------------------------------------------------------------
\bibliography{Bibliography} % build bibliography
\bibliographystyle{natdin} % DIN-style of the bibliography

\include{Erklaerung} % Selbständigkeitserklärung

% Appendix -----------------------------------------------------------------------
%   the contents of the appendix are included same as the other chapters,
%   by using the file "Appendix.tex".
% ------------------------------------------------------------------------------
\begin{appendix}
    \clearpage
    \pagenumbering{roman}
    \chapter{Anhang}
    \label{sec:Anhang}
    % adjust the margin of the listings in tables
    \setdefaultleftmargin{1em}{}{}{}{}{}
    \input{Appendix}
\end{appendix}

% Index ------------------------------------------------------------------------
%   to create an index, uncomment the following line
% ------------------------------------------------------------------------------
%\printindex

\end{document}
